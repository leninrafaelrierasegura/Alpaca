format_market_cap <- function(x) {
  sapply(x, function(val) {
    if (is.na(val)) {
      return("NA")
    }
    
    val <- as.numeric(val)
    
    if (val >= 1e12) {
      return(sprintf("%.1fT", val / 1e12))
    } else if (val >= 1e9) {
      return(sprintf("%.1fB", val / 1e9))
    } else if (val >= 1e6) {
      return(sprintf("%.1fM", val / 1e6))
    } else if (val >= 1e3) {
      return(sprintf("%.1fk", val / 1e3))
    } else {
      return(as.character(val))
    }
  })
}

Filters

Let X_k(t_i) denote the value of a stock X_k at day t_i, where i= 0,...,N. Let pct X_k(t_i) denote the percentage change of X_k(t_i) with respect to X_k(t_0), i.e., pct X_k(t_i) = (X_k(t_i) - X_k(t_0))/X_k(t_0) * 100. For each stock X_k, the filters are as follows:

  • The first filter is that X_k(t_0) <= X_k(t_N), i.e., the stock must have a positive growth trend within the period of analysis.

More stuff

Suppose a stock has price \(P_{\mathrm{start}}\) at time \(t_{\mathrm{start}}\) and price \(P_{\mathrm{end}}\) at time \(t_{\mathrm{end}}\), with \(\Delta t=t_{\mathrm{end}}-t_{\mathrm{start}}\) measured in days. There are several natural ways to quantify the speed at which the stock has grown.

The simplest measure is the absolute slope, \[ m_{\mathrm{abs}} = \frac{P_{\mathrm{end}}-P_{\mathrm{start}}}{\Delta t}, \] which measures the average increase in price per day. While intuitive, this measure depends strongly on the stock’s initial price, so expensive stocks tend to have larger values even if their relative growth is modest.

To remove this dependence, one can normalize by the initial price and use the fractional slope, \[ m_{\mathrm{frac}} = \frac{P_{\mathrm{end}}-P_{\mathrm{start}}} {P_{\mathrm{start}}\Delta t}, \] which measures the average fractional (or percentage) increase per day. This metric allows a fair comparison between stocks with different price levels.

A more natural measure from a financial perspective is the log-growth rate, \[ m_{\log} = \frac{\log(P_{\mathrm{end}}/P_{\mathrm{start}})} {\Delta t}, \] which assumes prices grow multiplicatively rather than linearly. If the stock followed an exponential growth model, \[ P(t)=P_{\mathrm{start}}e^{m_{\log}t}, \] then \(m_{\log}\) would be the exact continuous growth rate. Moreover, ranking stocks by \(m_{\log}\) is equivalent to ranking them by their compound annual growth rate (CAGR), making it a widely used measure in quantitative finance.

  arrange(date, .by_group = TRUE) |>

slope_comp <- stocks_data_grow |>
  filter(is.na(comment)) |>
  group_by(symbol) |>
  arrange(date, .by_group = TRUE) |>
  summarise(
    last_date  = last(date),
    last_price = last(lastsale),
    first_date = first(date),
    first_price = first(lastsale),

    half_date = {
      threshold <- last(lastsale) / 2
      idx <- which(lastsale <= threshold)
      if (length(idx) == 0) as.Date(NA) else max(date[idx])
    },

    days_to_double = as.numeric(last_date - half_date),
    days_from_start = as.numeric(last_date - first_date),

    country  = last(country),
    sector   = last(sector),
    industry = last(industry),
    name     = last(name),
    marketCap = format_market_cap(last(marketCap)),
    ipoyear   = last(ipoyear),

    .groups = "drop"
  ) |>
  mutate(half_price = last_price / 2) |>
  filter(!is.na(half_date)) |>
  arrange(days_to_double) |> 
  mutate(frac_slope = (last_price - first_price) / (first_price * days_from_start)) |>
  mutate(log_slope = log(last_price / first_price) / days_from_start) |>
  select(-last_date, -first_date, -half_date, -half_price, -days_from_start)|>
  relocate(marketCap, frac_slope, log_slope, ipoyear, name, .after = days_to_double)
